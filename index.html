async function loadPublicData() { const statusEl = document.getElementById('status'); const listEl = document.getElementById('hosp-list'); const HA_URL = "https://www.ha.org.hk/opendata/aewaitingtime/aewaitingtime-tc.json"; // 定義三個不同的中轉站，增加穩定性 const proxies = [ `https://api.allorigins.win/get?url=${encodeURIComponent(HA_URL)}`, `https://corsproxy.io/?${encodeURIComponent(HA_URL)}`, `https://thingproxy.freeboard.io/fetch/${HA_URL}` ]; for (let url of proxies) { try { console.log("嘗試連線路徑:", url); const response = await fetch(url); if (!response.ok) continue; // 如果失敗，嘗試下一個 const rawData = await response.json(); // 處理 AllOrigins 格式 (數據在 .contents) 或 其他直接回傳 JSON 嘅格式 const dataStr = rawData.contents ? rawData.contents : JSON.stringify(rawData); const data = typeof dataStr === 'string' ? JSON.parse(dataStr) : dataStr; if (data && data.waitTime) { renderTable(data); statusEl.innerText = "✅ 數據已透過備用路徑同步成功"; statusEl.className = "status-bar success"; return; // 成功後跳出循環 } } catch (err) { console.warn("目前路徑失效，切換中..."); } } statusEl.innerText = "⚠️ 所有 API 連線繁忙，請幾秒後重整。"; statusEl.className = "status-bar error"; } function renderTable(data) { const listEl = document.getElementById('hosp-list'); listEl.innerHTML = ""; data.waitTime.forEach(h => { const row = ` ${h.hospName}最後更新: ${data.updateTime} ${h.topWait} `; listEl.innerHTML += row; }); }
