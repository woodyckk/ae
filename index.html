// 3. 自動抓取仁安醫院並計算時間差 async function fetchUnion() { const url = "https://www.union.org/tc/"; // 仁安主頁 try { const res = await fetch(PROXY + encodeURIComponent(url)); const html = await res.text(); // 使用正規表達式提取「醫生現時處理」的時間 (格式如 00:27) const timeMatch = html.match(/(\d{2}:\d{2})/); if (timeMatch) { const processingTimeStr = timeMatch[1]; const now = new Date(); // 解析網頁上的時間 const [pHours, pMinutes] = processingTimeStr.split(':').map(Number); const pDate = new Date(); pDate.setHours(pHours, pMinutes, 0); // 計算分鐘差 (處理跨日情況) let diff = Math.floor((now - pDate) / 1000 / 60); if (diff < 0) diff += 1440; // 如果跨午夜，加 24 小時 // 顯示邏輯：如果差距很小，顯示預計等候時間 const waitEstimate = diff + 5; // 額外加 5 分鐘緩衝 document.getElementById('w-union').innerText = `約 ${waitEstimate} 分鐘`; document.getElementById('t-union').innerText = `網頁顯示處理至: ${processingTimeStr}`; } else { document.getElementById('w-union').innerText = "約 15-30 分鐘"; } } catch (e) { document.getElementById('w-union').innerText = "需查閱官網"; } }
